28 void bubbleSort(T &a, int N) {
29  bool swapp = true;
30  while (swapp) {
31  swapp = false;
32  for (size_t i = 0; i < N - 1; i++) {
33  if (a[i] > a[i + 1]) {
34  a[i] += a[i + 1];
35  a[i + 1] = a[i] - a[i + 1];
36  a[i] -= a[i + 1];
37  swapp = true;
38  }
39  }
40  }
